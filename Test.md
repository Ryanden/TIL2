function generatePermutationNumber(array) {
    let res = [];

    for (let i = 0; i < array.length; i++) {
      let rest = generatePermutationNumber(array.slice(0, i).concat(array.slice(i + 1)));

      if(!rest.length) {
        res.push([array[i]])

        res.push([array[i]])
